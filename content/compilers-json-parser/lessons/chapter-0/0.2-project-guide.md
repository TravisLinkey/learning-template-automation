## 1. Define Token Types

**Goal:** Create token type enum and token struct.

**Why it matters:** The lexer produces tokens, the parser consumes them. You need a standard token representation.

**Hints:**
- Create `TokenType` enum with all JSON tokens
- Create `Token` struct with type, value, line, column
- Include an `END` token for end of input

**Try it:** Define the token types.

<details>
<summary>Show Solution</summary>

```c
typedef enum {
    TOKEN_LBRACE,    // {
    TOKEN_RBRACE,    // }
    TOKEN_LBRACKET,  // [
    TOKEN_RBRACKET,  // ]
    TOKEN_COLON,     // :
    TOKEN_COMMA,     // ,
    TOKEN_STRING,    // "..."
    TOKEN_NUMBER,    // 123, 3.14
    TOKEN_TRUE,      // true
    TOKEN_FALSE,     // false
    TOKEN_NULL,      // null
    TOKEN_END,       // end of input
    TOKEN_ERROR,     // error
} TokenType;

typedef struct {
    TokenType type;
    char *value;  // for STRING and NUMBER
    int line;
    int column;
} Token;
```

## 2. Implement the Lexer

**Goal:** Convert JSON text into a stream of tokens.

**Why it matters:** The lexer is the first phase of any parser. It handles character-level details so the parser can work at the token level.

**Hints:**
- Use a position index into the input string
- Skip whitespace (space, tab, newline)
- For strings: scan until closing `"`, handle escapes
- For numbers: scan digits, optional decimal, optional exponent
- For literals: check for "true", "false", "null"
- Track line/column for error reporting

**Try it:** Implement a basic lexer.

<details>
<parameter name>Show Solution</summary>

```c
typedef struct {
    const char *input;
    size_t pos;
    int line;
    int column;
} Lexer;

static void lexer_init(Lexer *lex, const char *input) {
    lex->input = input;
    lex->pos = 0;
    lex->line = 1;
    lex->column = 1;
}

static char peek(Lexer *lex) {
    return lex->input[lex->pos];
}

static char next(Lexer *lex) {
    char c = lex->input[lex->pos++];
    if (c == '\n') {
        lex->line++;
        lex->column = 1;
    } else {
        lex->column++;
    }
    return c;
}

static void skip_whitespace(Lexer *lex) {
    while (peek(lex) == ' ' || peek(lex) == '\t' || 
           peek(lex) == '\n' || peek(lex) == '\r') {
        next(lex);
    }
}

static Token make_token(Lexer *lex, TokenType type, char *value) {
    Token token;
    token.type = type;
    token.value = value;
    token.line = lex->line;
    token.column = lex->column;
    return token;
}

static Token next_token(Lexer *lex) {
    skip_whitespace(lex);
    
    char c = peek(lex);
    if (c == '\0') {
        return make_token(lex, TOKEN_END, NULL);
    }
    next(lex);
    
    switch (c) {
        case '{': return make_token(lex, TOKEN_LBRACE, NULL);
        case '}': return make_token(lex, TOKEN_RBRACE, NULL);
        case '[': return make_token(lex, TOKEN_LBRACKET, NULL);
        case ']': return make_token(lex, TOKEN_RBRACKET, NULL);
        case ':': return make_token(lex, TOKEN_COLON, NULL);
        case ',': return make_token(lex, TOKEN_COMMA, NULL);
        case '"': return lex_string(lex);
        default:
            if (isdigit(c) || c == '-') return lex_number(lex, c);
            return lex_literal(lex, c);
    }
}
```

## 3. Define AST Nodes

**Goal:** Create data structures to represent parsed JSON.

**Why it matters:** The parser builds a tree of nodes. Each node represents a JSON value (object, array, string, etc.).

**Hints:**
- Use enum for node type
- Each node stores its type and value
- Objects store key-value pairs (children + keys)
- Arrays store child nodes in order

**Try it:** Define AST node structures.

<details>
<summary>Show Solution</summary>

```c
typedef enum {
    NODE_OBJECT,
    NODE_ARRAY,
    NODE_STRING,
    NODE_NUMBER,
    NODE_BOOLEAN,
    NODE_NULL,
} NodeType;

typedef struct Node {
    NodeType type;
    union {
        char *string_value;
        double number_value;
        int bool_value;
    } value;
    
    // For objects and arrays
    struct Node **children;
    size_t child_count;
    
    // For objects (key-value pairs)
    char **keys;
    size_t key_count;
} Node;

Node *node_new(NodeType type) {
    Node *node = calloc(1, sizeof(Node));
    node->type = type;
    return node;
}
```

## 4. Implement the Parser

**Goal:** Parse token stream into AST.

**Why it matters:** The parser applies grammar rules to produce a structured AST. For JSON, we use recursive descent.

**Hints:**
- `parse_value()` is the entry point
- For `{`, parse as object: loop through key-value pairs
- For `[`, parse as array: loop through values
- For other tokens, parse as primitives (string, number, boolean, null)
- Use recursion for nested objects/arrays
- Check for expected token types, advance on success

**Try it:** Implement recursive descent parser.

<details>
<summary>Show Solution</summary>

```c
typedef struct {
    Lexer lexer;
    Token current;
} Parser;

static void parser_init(Parser *parser, const char *input) {
    lexer_init(&parser->lexer, input);
    parser->current = next_token(&parser->lexer);
}

static void advance(Parser *parser) {
    parser->current = next_token(&parser->lexer);
}

static void expect(Parser *parser, TokenType type) {
    if (parser->current.type != type) {
        fprintf(stderr, "Parse error: expected %d but got %d at line %d\n",
                type, parser->current.type, parser->current.line);
        exit(1);
    }
    advance(parser);
}

static Node *parse_value(Parser *parser) {
    switch (parser->current.type) {
        case TOKEN_LBRACE: return parse_object(parser);
        case TOKEN_LBRACKET: return parse_array(parser);
        case TOKEN_STRING: return parse_string(parser);
        case TOKEN_NUMBER: return parse_number(parser);
        case TOKEN_TRUE:
        case TOKEN_FALSE: return parse_boolean(parser);
        case TOKEN_NULL: return parse_null(parser);
        default:
            fprintf(stderr, "Unexpected token: %d\n", parser->current.type);
            exit(1);
    }
}

static Node *parse_object(Parser *parser) {
    expect(parser, TOKEN_LBRACE);
    
    Node *obj = node_new(NODE_OBJECT);
    
    if (parser->current.type != TOKEN_RBRACE) {
        while (1) {
            // Parse key
            expect(parser, TOKEN_STRING);
            char *key = parser->current.value;  // Previous string value
            advance(parser);
            
            expect(parser, TOKEN_COLON);
            
            // Parse value
            Node *value = parse_value(parser);
            
            // Add to object
            obj->keys = realloc(obj->keys, (obj->key_count + 1) * sizeof(char*));
            obj->keys[obj->key_count] = key;
            obj->children = realloc(obj->children, (obj->child_count + 1) * sizeof(Node*));
            obj->children[obj->child_count] = value;
            obj->key_count++;
            obj->child_count++;
            
            if (parser->current.type == TOKEN_COMMA) {
                advance(parser);
            } else {
                break;
            }
        }
    }
    
    expect(parser, TOKEN_RBRACE);
    return obj;
}

static Node *parse_array(Parser *parser) {
    expect(parser, TOKEN_LBRACKET);
    
    Node *arr = node_new(NODE_ARRAY);
    
    while (parser->current.type != TOKEN_RBRACKET) {
        Node *value = parse_value(parser);
        arr->children = realloc(arr->children, (arr->child_count + 1) * sizeof(Node*));
        arr->children[arr->child_count++] = value;
        
        if (parser->current.type == TOKEN_COMMA) {
            advance(parser);
        } else {
            break;
        }
    }
    
    expect(parser, TOKEN_RBRACKET);
    return arr;
}
```

(Additional functions for primitives would follow similar patterns.)

## 5. Implement Tree Traversal

**Goal:** Visit all nodes in the AST.

**Why it matters:** To do anything useful with the parsed JSON (print, transform, query), you need to traverse the tree.

**Hints:**
- Use depth-first traversal
- Callback function called for each node
- Can pass user data to callback
- Handle different node types appropriately

**Try it:** Implement traversal function.

<details>
<summary>Show Solution</summary>

```c
typedef void (*NodeVisitor)(Node *node, int depth, void *user_data);

static void traverse(Node *node, NodeVisitor visitor, int depth, void *user_data) {
    if (node == NULL) return;
    
    visitor(node, depth, user_data);
    
    for (size_t i = 0; i < node->child_count; i++) {
        traverse(node->children[i], visitor, depth + 1, user_data);
    }
}

static void print_node(Node *node, int depth, void *user_data) {
    for (int i = 0; i < depth; i++) printf("  ");
    
    switch (node->type) {
        case NODE_OBJECT:
            printf("Object\n");
            break;
        case NODE_ARRAY:
            printf("Array\n");
            break;
        case NODE_STRING:
            printf("String: %s\n", node->value.string_value);
            break;
        case NODE_NUMBER:
            printf("Number: %f\n", node->value.number_value);
            break;
        case NODE_BOOLEAN:
            printf("Boolean: %s\n", node->value.bool_value ? "true" : "false");
            break;
        case NODE_NULL:
            printf("Null\n");
            break;
    }
}

// Usage:
traverse(root, print_node, 0, NULL);
```

## 6. Putting It Together

**Goal:** Combine lexer, parser, and traversal into a usable JSON parser.

**Try it:** Create the final API.

<details>
<summary>Show Solution</summary>

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// ... all previous code ...

Node *json_parse(const char *input) {
    Parser parser;
    parser_init(&parser, input);
    return parse_value(&parser);
}

void json_traverse(Node *root, NodeVisitor visitor, void *user_data) {
    traverse(root, visitor, 0, user_data);
}

void json_free(Node *root) {
    if (root == NULL) return;
    for (size_t i = 0; i < root->child_count; i++) {
        json_free(root->children[i]);
    }
    free(root->children);
    free(root->keys);
    free(root);
}

// Example usage
int main(void) {
    const char *json = "{\"name\": \"Alice\", \"age\": 30}";
    
    Node *root = json_parse(json);
    
    printf("Parsed JSON:\n");
    json_traverse(root, print_node, NULL);
    
    json_free(root);
    return 0;
}
```

## 7. Final Tips

- **Incremental development** — Get lexer working first, test with print statements
- **Error handling** — Add meaningful error messages with line/column
- **Memory management** — Track all allocations, free in json_free()
- **Test cases** — Test empty objects {}, empty arrays [], nested structures
- **Extend** — Add more node types, query functions, or transform utilities

If your parser can handle complex JSON like `{"data": [{"id": 1}, {"id": 2}]}`, you've built a working JSON parser!
