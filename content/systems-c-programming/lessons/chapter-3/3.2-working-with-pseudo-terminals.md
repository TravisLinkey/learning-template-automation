# 3.2 — Working with Pseudo-Terminals

This lesson covers practical PTY operations for building a terminal emulator.

## Complete PTY Setup

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <pty.h>
#include <utmp.h>
#include <fcntl.h>

int main(void) {
    int master, slave;
    char *slave_name;
    struct termios slave_termios;
    struct winsize slave_winsize;
    
    // Open PTY
    if (openpty(&master, &slave, NULL, NULL, NULL) < 0) {
        perror("openpty");
        return 1;
    }
    
    // Get slave name
    slave_name = ptsname(master);
    printf("PTY opened: master=%d, slave=%d (%s)\n", master, slave, slave_name);
    
    // Close file descriptors for now
    close(master);
    close(slave);
    
    return 0;
}
```

## Spawning a Shell in PTY

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <pty.h>
#include <sys/wait.h>

int spawn_shell(int *master_fd) {
    int slave, master;
    pid_t pid;
    char *slave_name;
    
    // Open PTY
    if (openpty(&master, &slave, NULL, NULL, NULL) < 0) {
        return -1;
    }
    
    pid = fork();
    
    if (pid < 0) {
        close(master);
        close(slave);
        return -1;
    }
    
    if (pid == 0) {
        // Child process
        close(master);
        
        // Create new session
        setsid();
        
        // Set as controlling terminal
        ioctl(slave, TIOCSCTTY, 0);
        
        // Redirect stdin, stdout, stderr to slave
        dup2(slave, 0);
        dup2(slave, 1);
        dup2(slave, 2);
        close(slave);
        
        // Execute shell
        execl("/bin/bash", "bash", NULL);
        _exit(1);
    }
    
    // Parent process
    close(slave);
    *master_fd = master;
    
    return pid;  // Return child's PID
}
```

## Reading and Writing PTY

### Non-blocking Read

```c
#include <fcntl.h>

// Set non-blocking
int flags = fcntl(fd, F_GETFL, 0);
fcntl(fd, F_SETFL, flags | O_NONBLOCK);

// Read with select()
fd_set readfds;
FD_ZERO(&readfds);
FD_SET(master_fd, &readfds);

struct timeval tv = {1, 0};  // 1 second timeout
int ret = select(master_fd + 1, &readfds, NULL, NULL, &tv);

if (ret > 0 && FD_ISSET(master_fd, &readfds)) {
    char buf[4096];
    ssize_t n = read(master_fd, buf, sizeof(buf));
    // Process output
}
```

### Write to PTY

```c
const char *input = "ls -la\n";
write(master_fd, input, strlen(input));
```

## select() for Multiplexing

```c
#include <sys/select.h>

int select(int nfds, fd_set *readfds, fd_set *writefds, fd_set *exceptfds, struct timeval *timeout);

// Helper macros
FD_ZERO(&readfds);
FD_SET(STDIN_FILENO, &readfds);
FD_SET(master_fd, &readfds);

int maxfd = (master_fd > STDIN_FILENO) ? master_fd : STDIN_FILENO;
int ret = select(maxfd + 1, &readfds, NULL, NULL, NULL);

if (ret > 0) {
    if (FD_ISSET(STDIN_FILENO, &readfds)) {
        // Read from user, write to PTY
        char buf[4096];
        ssize_t n = read(STDIN_FILENO, buf, sizeof(buf));
        write(master_fd, buf, n);
    }
    if (FD_ISSET(master_fd, &readfds)) {
        // Read from PTY, write to user
        char buf[4096];
        ssize_t n = read(master_fd, buf, sizeof(buf));
        write(STDOUT_FILENO, buf, n);
    }
}
```

## PTY with Custom Terminal Settings

```c
#include <termios.h>

int spawn_shell_with_settings(int *master_fd) {
    int master, slave;
    pid_t pid;
    struct termios termios;
    struct winsize winsize;
    
    // Get current terminal settings
    tcgetattr(STDIN_FILENO, &termios);
    ioctl(STDIN_FILENO, TIOCGWINSZ, &winsize);
    
    // Open PTY with custom settings
    if (openpty(&master, &slave, NULL, &termios, &winsize) < 0) {
        return -1;
    }
    
    pid = fork();
    
    if (pid == 0) {
        close(master);
        setsid();
        ioctl(slave, TIOCSCTTY, 0);
        
        dup2(slave, 0);
        dup2(slave, 1);
        dup2(slave, 2);
        close(slave);
        
        execl("/bin/bash", "bash", NULL);
        _exit(1);
    }
    
    close(slave);
    *master_fd = master;
    return pid;
}
```

## Window Resize Handling

```c
#include <sys/ioctl.h>
#include <signal.h>
#include <signal.h>

void handle_winch(int sig) {
    struct winsize winsize;
    ioctl(STDIN_FILENO, TIOCGWINSZ, &winsize);
    // Send to PTY master
    ioctl(master_fd, TIOCSWINSZ, &winsize);
}

signal(SIGWINCH, handle_winch);
```

## Complete Simple Terminal Emulator

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <pty.h>
#include <sys/select.h>
#include <sys/wait.h>
#include <termios.h>

int main(void) {
    int master;
    pid_t pid = spawn_shell(&master);
    
    if (pid < 0) {
        perror("spawn_shell");
        return 1;
    }
    
    // Set raw mode on stdin
    struct termios orig_termios;
    tcgetattr(STDIN_FILENO, &orig_termios);
    struct termios raw = orig_termios;
    cfmakeraw(&raw);
    tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw);
    
    // Main loop: select between stdin and PTY
    while (1) {
        fd_set readfds;
        FD_ZERO(&readfds);
        FD_SET(STDIN_FILENO, &readfds);
        FD_SET(master, &readfds);
        
        int maxfd = (master > STDIN_FILENO) ? master : STDIN_FILENO;
        int ret = select(maxfd + 1, &readfds, NULL, NULL, NULL);
        
        if (ret < 0) break;
        
        // User input -> PTY
        if (FD_ISSET(STDIN_FILENO, &readfds)) {
            char buf[4096];
            ssize_t n = read(STDIN_FILENO, buf, sizeof(buf));
            if (n > 0) write(master, buf, n);
        }
        
        // PTY output -> stdout
        if (FD_ISSET(master, &readfds)) {
            char buf[4096];
            ssize_t n = read(master, buf, sizeof(buf));
            if (n <= 0) break;
            write(STDOUT_FILENO, buf, n);
        }
    }
    
    // Cleanup
    tcsetattr(STDIN_FILENO, TCSAFLUSH, &orig_termios);
    close(master);
    waitpid(pid, NULL, 0);
    
    return 0;
}
```
