# 3.3 — Terminal Control and Signals

This lesson covers terminal control and signal handling for robust terminal applications.

## Terminal Control Modes

### Canonical vs Non-canonical

```c
struct termios {
    tcflag_t c_lflag;  // Local modes
    // ...
};

// Canonical mode (line editing enabled)
struct termios t;
tcgetattr(fd, &t);
t.c_lflag |= ICANON;  // Canonical
tcsetattr(fd, TCSAFLUSH, &t);

// Non-canonical (immediate input)
t.c_lflag &= ~ICANON;  // Non-canonical
tcsetattr(fd, TCSAFLUSH, &t);
```

### Echo

```c
// Enable echo
t.c_lflag |= ECHO;

// Disable echo
t.c_lflag &= ~ECHO;
```

### Common Flags

```c
// Input modes
c_iflag:
// IGNBRK = ignore break
// BRKINT = signal on break
// ICRNL = map CR to NL
// IXON = enable XON/XOFF

// Output modes
c_oflag:
// OPOST = output processing
// ONLCR = map NL to CR-NL

// Control modes
c_cflag:
// CSIZE = character size
// CS8 = 8-bit
// CREAD = enable receiver

// Local modes
c_lflag:
// ISIG = enable signals
// ICANON = canonical
// ECHO = echo
// ECHOE = echo erase
// ECHOK = echo NL
```

## cfmakeraw() — Quick Raw Mode

```c
#include <termios.h>

void cfmakeraw(struct termios *termios_p);
```

Sets sensible defaults for raw mode:
- Disable input processing
- Disable output processing  
- Disable character size mask
- Enable receiver
- Disable signals and echo

```c
struct termios t;
tcgetattr(STDIN_FILENO, &t);
cfmakeraw(&t);
tcsetattr(STDIN_FILENO, TCSAFLUSH, &t);
```

## Signal Handling

### Common Signals

| Signal | Default | Description |
|--------|---------|-------------|
| `SIGINT` | Terminate | Ctrl+C |
| `SIGQUIT` | Core dump | Ctrl+\\ |
| `SIGTSTP` | Stop | Ctrl+Z |
| `SIGTTIN` | Stop | Background read |
| `SIGTTOU` | Stop | Background write |
| `SIGWINCH` | Ignore | Window resize |
| `SIGCHLD` | Ignore | Child exited |

### Setting Up Signal Handlers

```c
#include <signal.h>

void handle_signal(int sig) {
    // Handle signal
}

signal(SIGINT, handle_signal);
```

Or use POSIX sigaction:

```c
#include <signal.h>

struct sigaction sa;
sa.sa_handler = handle_signal;
sigemptyset(&sa.sa_mask);
sa.sa_flags = 0;
sigaction(SIGINT, &sa, NULL);
```

### Signal for Clean Shutdown

```c
volatile sig_atomic_t keep_running = 1;

void signal_handler(int sig) {
    keep_running = 0;
}

int main(void) {
    signal(SIGINT, signal_handler);
    signal(SIGTERM, signal_handler);
    
    while (keep_running) {
        // Main loop
    }
    
    // Cleanup
    return 0;
}
```

## Proper Cleanup on Exit

```c
#include <stdlib.h>

void cleanup(void) {
    // Restore terminal
    tcsetattr(STDIN_FILENO, TCSAFLUSH, &orig_termios);
    
    // Close PTY
    if (master_fd >= 0) close(master_fd);
    
    // Wait for child
    if (child_pid > 0) waitpid(child_pid, NULL, 0);
}

int main(void) {
    atexit(cleanup);
    
    // ... main code ...
}
```

## Window Size Changes

### Detecting Resize

```c
#include <signal.h>
#include <sys/ioctl.h>

struct winsize winsize;

signal(SIGWINCH, handle_winch);

void handle_winch(int sig) {
    ioctl(STDIN_FILENO, TIOCGWINSZ, &winsize);
    // Update PTY window size
    ioctl(master_fd, TIOCSWINSZ, &winsize);
}
```

### Sending Resize to PTY

```c
void set_window_size(int fd, struct winsize *ws) {
    ioctl(fd, TIOCSWINSZ, ws);
}
```

## Escape Sequences

Terminals use ANSI escape sequences for control:

```c
// Colors
#define ESC "\033"
#define RESET ESC "[0m"
#define RED ESC "[31m"
#define GREEN ESC "[32m"

// Cursor movement
#define CURSOR_HOME ESC "[H"
#define CURSOR_UP(n) ESC "[%dA"
#define CURSOR_DOWN(n) ESC "[%dB"

// Clear screen
#define CLEAR ESC "[2J"

// Print with color
printf(RED "Error: " RESET "%s\n", message);
```

## Complete Signal-Safe Terminal

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <signal.h>
#include <termios.h>
#include <pty.h>
#include <sys/select.h>
#include <sys/wait.h>

static int master_fd = -1;
static pid_t child_pid = -1;
static struct termios orig_termios;
static volatile sig_atomic_t keep_running = 1;

void cleanup(void) {
    if (master_fd >= 0) close(master_fd);
    if (child_pid > 0) waitpid(child_pid, NULL, 0);
    tcsetattr(STDIN_FILENO, TCSAFLUSH, &orig_termios);
}

void signal_handler(int sig) {
    keep_running = 0;
}

int main(void) {
    // Setup signal handlers
    signal(SIGINT, signal_handler);
    signal(SIGTERM, signal_handler);
    signal(SIGCHLD, signal_handler);
    
    // Save terminal settings
    tcgetattr(STDIN_FILENO, &orig_termios);
    
    // Register cleanup
    atexit(cleanup);
    
    // Set raw mode
    struct termios raw = orig_termios;
    cfmakeraw(&raw);
    tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw);
    
    // Spawn shell in PTY
    openpty(&master, &slave, NULL, NULL, NULL);
    child_pid = fork();
    
    if (child_pid == 0) {
        close(master);
        setsid();
        ioctl(slave, TIOCSCTTY, 0);
        dup2(slave, 0);
        dup2(slave, 1);
        dup2(slave, 2);
        close(slave);
        execl("/bin/bash", "bash", NULL);
        _exit(1);
    }
    
    close(slave);
    
    // Main loop
    while (keep_running) {
        fd_set readfds;
        FD_ZERO(&readfds);
        FD_SET(STDIN_FILENO, &readfds);
        FD_SET(master, &readfds);
        
        int maxfd = (master > STDIN_FILENO) ? master : STDIN_FILENO;
        int ret = select(maxfd + 1, &readfds, NULL, NULL, NULL);
        
        if (ret <= 0) continue;
        
        if (FD_ISSET(STDIN_FILENO, &readfds)) {
            char buf[4096];
            ssize_t n = read(STDIN_FILENO, buf, sizeof(buf));
            if (n > 0) write(master, buf, n);
        }
        
        if (FD_ISSET(master, &readfds)) {
            char buf[4096];
            ssize_t n = read(master, buf, sizeof(buf));
            if (n <= 0) break;
            write(STDOUT_FILENO, buf, n);
        }
    }
    
    return 0;
}
```
