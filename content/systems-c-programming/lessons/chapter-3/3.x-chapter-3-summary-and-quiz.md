## Question 1: PTY Creation

**Task:** Write code to create a PTY pair using the simplest method.

**Expected Output:**
```
PTY created: master=3, slave=4 (/dev/pts/1)
```
(Actual numbers may vary)

<details>
<summary>Show Hint</summary>

Use the `openpty()` function from `<pty.h>`.

</details>

<details>
<summary>Show Solution</summary>

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <pty.h>

int main(void) {
    int master, slave;
    char *slave_name;
    
    if (openpty(&master, &slave, NULL, NULL, NULL) < 0) {
        perror("openpty failed");
        return 1;
    }
    
    slave_name = ptsname(master);
    printf("PTY created: master=%d, slave=%d (%s)\n", 
           master, slave, slave_name);
    
    close(master);
    close(slave);
    return 0;
}
```

</details>

## Question 2: Terminal Modes

**Task:** Write code to set the terminal to raw mode using cfmakeraw().

**Input:**
```c
// Current terminal is in canonical mode with echo
```

**Expected Output:**
```
Terminal set to raw mode
```

<details>
<summary>Show Hint</summary>

Use tcgetattr() to get current settings, cfmakeraw() to modify, and tcsetattr() to apply.

</details>

<details>
<summary>Show Solution</summary>

```c
#include <stdio.h>
#include <termios.h>
#include <unistd.h>

int main(void) {
    struct termios t;
    
    // Get current settings
    if (tcgetattr(STDIN_FILENO, &t) < 0) {
        perror("tcgetattr");
        return 1;
    }
    
    // Apply raw mode
    cfmakeraw(&t);
    
    // Set new settings
    if (tcsetattr(STDIN_FILENO, TCSAFLUSH, &t) < 0) {
        perror("tcsetattr");
        return 1;
    }
    
    printf("Terminal set to raw mode\n");
    return 0;
}
```

</details>

## Question 3: Signal Handler Setup

**Task:** Write a signal handler that sets a flag when SIGINT is received.

**Expected Output:**
```
Signal handler installed for SIGINT
```

<details>
<summary>Show Hint</summary>

Use a `volatile sig_atomic_t` variable and the `signal()` function.

</details>

<details>
<summary>Show Solution</summary>

```c
#include <stdio.h>
#include <signal.h>
#include <unistd.h>

volatile sig_atomic_t keep_running = 1;

void handle_signal(int sig) {
    keep_running = 0;
}

int main(void) {
    signal(SIGINT, handle_signal);
    
    printf("Signal handler installed for SIGINT\n");
    
    // Test it
    printf("Waiting for Ctrl+C...\n");
    while (keep_running) {
        sleep(1);
    }
    
    printf("Received SIGINT, exiting cleanly\n");
    return 0;
}
```

</details>

## Question 4: Simple PTY Shell

**Task:** Write code that spawns a shell in a PTY and echoes output from the shell to stdout.

**Hint:** Fork, create PTY, set up slave as controlling terminal, execute bash.

<details>
<summary>Show Hint</summary>

Combine fork(), openpty(), setsid(), ioctl(TIOCSCTTY), and execl().

</details>

<details>
<summary>Show Solution</summary>

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <pty.h>
#include <sys/select.h>
#include <sys/wait.h>
#include <termios.h>

int main(void) {
    int master, slave;
    pid_t pid;
    
    // Open PTY
    if (openpty(&master, &slave, NULL, NULL, NULL) < 0) {
        perror("openpty");
        return 1;
    }
    
    pid = fork();
    
    if (pid < 0) {
        perror("fork");
        return 1;
    }
    
    if (pid == 0) {
        // Child: set up PTY slave
        close(master);
        setsid();
        ioctl(slave, TIOCSCTTY, 0);
        
        dup2(slave, 0);
        dup2(slave, 1);
        dup2(slave, 2);
        close(slave);
        
        execl("/bin/bash", "bash", NULL);
        _exit(1);
    }
    
    // Parent: read from PTY and echo to stdout
    close(slave);
    
    char buf[4096];
    ssize_t n;
    while ((n = read(master, buf, sizeof(buf))) > 0) {
        write(STDOUT_FILENO, buf, n);
    }
    
    close(master);
    waitpid(pid, NULL, 0);
    
    return 0;
}
```

</details>

## Question 5: Cleanup Function

**Task:** Write an atexit cleanup function that restores terminal settings.

**Hint:** Use a static variable to store original termios.

<details>
<summary>Show Hint</summary>

Save original termios at startup, then register cleanup with atexit().

</details>

<details>
<summary>Show Solution</summary>

```c
#include <stdio.h>
#include <stdlib.h>
#include <termios.h>
#include <unistd.h>

static struct termios orig_termios;
static int master_fd = -1;

void cleanup(void) {
    // Restore terminal settings
    tcsetattr(STDIN_FILENO, TCSAFLUSH, &orig_termios);
    
    // Close PTY if open
    if (master_fd >= 0) {
        close(master_fd);
    }
    
    printf("\nCleanup complete\n");
}

int main(void) {
    // Save original terminal settings
    tcgetattr(STDIN_FILENO, &orig_termios);
    
    // Register cleanup
    atexit(cleanup);
    
    // Set raw mode (simulating normal operation)
    struct termios raw = orig_termios;
    cfmakeraw(&raw);
    tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw);
    
    printf("Press Enter to exit and see cleanup...\n");
    getchar();
    
    return 0;
}
```

</details>
