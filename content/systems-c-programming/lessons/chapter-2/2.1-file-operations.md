# 2.1 — File Operations

File I/O is essential for persistent storage. This lesson covers reading and writing files in C.

## Opening Files

```c
FILE *fopen(const char *path, const char *mode);
```

| Mode | Description |
|------|-------------|
| `"r"` | Read (file must exist) |
| `"w"` | Write (creates/truncates) |
| `"a"` | Append (creates or adds to end) |
| `"r+"` | Read and write (file must exist) |
| `"w+"` | Read and write (creates/truncates) |
| `"a+"` | Append and read |

```c
FILE *fp = fopen("data.txt", "r");
if (fp == NULL) {
    perror("Failed to open file");
    return 1;
}
```

## Closing Files

```c
int fclose(FILE *stream);
```

```c
FILE *fp = fopen("file.txt", "r");
// ... use file ...
fclose(fp);
```

## Reading Files

### Character I/O

```c
int fgetc(FILE *stream);  // Returns int (EOF on error)
int fputc(int c, FILE *stream);

// Example: copy file character by character
int c;
while ((c = fgetc(fp)) != EOF) {
    fputc(c, stdout);
}
```

### Line I/O

```c
char *fgets(char *s, int size, FILE *stream);
int fputs(const char *s, FILE *stream);

// Read a line (up to size-1 characters)
char buffer[256];
if (fgets(buffer, sizeof(buffer), fp) != NULL) {
    // Process line
}
```

### Formatted I/O

```c
int fprintf(FILE *stream, const char *format, ...);
int fscanf(FILE *stream, const char *format, ...);

// Write formatted
fprintf(fp, "Name: %s, Age: %d\n", "Alice", 30);

// Read formatted
char name[50];
int age;
fscanf(fp, "%s %d", name, &age);
```

### Binary I/O

```c
size_t fread(void *ptr, size_t size, size_t nmemb, FILE *stream);
size_t fwrite(const void *ptr, size_t size, size_t nmemb, FILE *stream);
```

```c
// Write array to file
int data[] = {1, 2, 3, 4, 5};
fwrite(data, sizeof(int), 5, fp);

// Read array from file
int read_data[5];
fread(read_data, sizeof(int), 5, fp);
```

## Example: Copy File

```c
#include <stdio.h>
#include <stdlib.h>

int copy_file(const char *src, const char *dst) {
    FILE *in = fopen(src, "rb");
    if (in == NULL) {
        perror("Failed to open source");
        return 1;
    }
    
    FILE *out = fopen(dst, "wb");
    if (out == NULL) {
        perror("Failed to open destination");
        fclose(in);
        return 1;
    }
    
    // Copy character by character
    int c;
    while ((c = fgetc(in)) != EOF) {
        fputc(c, out);
    }
    
    fclose(in);
    fclose(out);
    return 0;
}

int main(int argc, char *argv[]) {
    if (argc != 3) {
        fprintf(stderr, "Usage: %s <source> <destination>\n", argv[0]);
        return 1;
    }
    return copy_file(argv[1], argv[2]);
}
```

## Example: Read Lines and Parse

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(void) {
    FILE *fp = fopen("data.txt", "r");
    if (fp == NULL) {
        perror("Failed to open file");
        return 1;
    }
    
    char line[256];
    while (fgets(line, sizeof(line), fp) != NULL) {
        // Remove newline
        line[strcspn(line, "\n")] = '\0';
        
        // Parse: name,age format
        char name[50];
        int age;
        if (sscanf(line, "%49[^,],%d", name, &age) == 2) {
            printf("Name: %s, Age: %d\n", name, age);
        }
    }
    
    fclose(fp);
    return 0;
}
```

## File Position

```c
int fseek(FILE *stream, long offset, int whence);
long ftell(FILE *stream);
void rewind(FILE *stream);
```

- `SEEK_SET` — from beginning
- `SEEK_CUR` — from current position
- `SEEK_END` — from end

```c
fseek(fp, 0, SEEK_END);      // Go to end
long size = ftell(fp);       // Get position (file size)
fseek(fp, 0, SEEK_SET);     // Go back to beginning
rewind(fp);                  // Equivalent to fseek(fp, 0, SEEK_SET)
```

## Error Handling

```c
// perror prints "prefix: error message"
FILE *fp = fopen("nonexistent.txt", "r");
if (fp == NULL) {
    perror("Failed to open file");  // "Failed to open file: No such file or directory"
    return 1;
}

// Check for errors during reading
if (ferror(fp)) {
    perror("Read error");
}
```

## Checking File Existence

```c
#include <stdio.h>

int file_exists(const char *path) {
    FILE *fp = fopen(path, "r");
    if (fp == NULL) {
        return 0;  // Doesn't exist
    }
    fclose(fp);
    return 1;  // Exists
}
```
