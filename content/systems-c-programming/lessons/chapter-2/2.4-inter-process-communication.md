# 2.4 — Inter-Process Communication

Processes often need to communicate. This lesson covers common IPC mechanisms in Unix/C.

## IPC Methods Overview

| Method | Description | Use Case |
|--------|-------------|----------|
| **Pipe** | Unidirectional byte stream | Parent-child communication |
| **FIFO** | Named pipe | Unrelated processes |
| **Signal** | Async notifications | Events, interrupts |
| **Shared Memory** | Common memory region | Fast data exchange |
| **Message Queue** | Message-based | Async communication |
| **Socket** | Network/local communication | Any client-server |

## Pipes

A pipe is a unidirectional communication channel:

```c
#include <unistd.h>

int pipe(int fds[2]);  // fds[0] = read, fds[1] = write
```

### Parent-Child Pipe Example

```c
#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>
#include <string.h>

int main(void) {
    int pipefd[2];
    pipe(pipefd);
    
    pid_t pid = fork();
    
    if (pid == 0) {
        // Child: write to pipe
        close(pipefd[0]);  // Close read end
        
        const char *msg = "Hello from child!\n";
        write(pipefd[1], msg, strlen(msg));
        
        close(pipefd[1]);  // Close write end
        _exit(0);
    } else {
        // Parent: read from pipe
        close(pipefd[1]);  // Close write end
        
        char buffer[256];
        ssize_t n = read(pipefd[0], buffer, sizeof(buffer));
        buffer[n] = '\0';
        
        printf("Parent received: %s", buffer);
        
        close(pipefd[0]);  // Close read end
        wait(NULL);
    }
    
    return 0;
}
```

## dup() — Duplicate File Descriptors

```c
#include <unistd.h>

int dup(int oldfd);
int dup2(int oldfd, int newfd);
```

Redirect stdin/stdout using dup2:

```c
#include <fcntl.h>
#include <unistd.h>

// Redirect stdout to file
int fd = open("output.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644);
dup2(fd, STDOUT_FILENO);
close(fd);

// Now printf goes to file!
printf("This goes to the file\n");
```

## FIFOs (Named Pipes)

Create a FIFO for unrelated processes to communicate:

```c
#include <sys/types.h>
#include <sys/stat.h>

mkfifo("/tmp/myfifo", 0666);
```

### Server (creates FIFO, reads):

```c
int fd = open("/tmp/myfifo", O_RDONLY);
read(fd, buffer, sizeof(buffer));
close(fd);
unlink("/tmp/myfifo");  // Clean up
```

### Client (writes):

```c
int fd = open("/tmp/myfifo", O_WRONLY);
write(fd, "message", 7);
close(fd);
```

## Signals for IPC

Signals can communicate events between processes:

```c
#include <signal.h>

// Child: setup signal handler
signal(SIGUSR1, handler);
pause();  // Wait for signal

// Parent: send signal
kill(child_pid, SIGUSR1);
```

### Signal-Handler Example

```c
#include <stdio.h>
#include <signal.h>
#include <unistd.h>

volatile sig_atomic_t got_signal = 0;

void handler(int sig) {
    got_signal = 1;
}

int main(void) {
    signal(SIGUSR1, handler);
    
    printf("Waiting for signal...\n");
    while (!got_signal) {
        sleep(1);
    }
    
    printf("Got signal!\n");
    return 0;
}
```

## Shared Memory

### shmget/shmat/shmdt

```c
#include <sys/shm.h>

// Create shared memory
int shmid = shmget(key, size, IPC_CREAT | 0666);

// Attach (get pointer)
void *shm = shmat(shmid, NULL, 0);

// Use shared memory
((int*)shm)[0] = 42;

// Detach
shmdt(shm);

// Remove
shmctl(shmid, IPC_RMID, NULL);
```

### Complete Shared Memory Example

```c
// writer.c
#include <stdio.h>
#include <sys/shm.h>
#include <sys/ipc.h>

int main(void) {
    key_t key = ftok("/tmp", 'S');
    int shmid = shmget(key, 1024, IPC_CREAT | 0666);
    char *shm = shmat(shmid, NULL, 0);
    
    sprintf(shm, "Hello from writer!");
    
    shmdt(shm);
    return 0;
}

// reader.c
#include <stdio.h>
#include <sys/shm.h>
#include <sys/ipc.h>

int main(void) {
    key_t key = ftok("/tmp", 'S');
    int shmid = shmget(key, 1024, 0666);
    char *shm = shmat(shmid, NULL, SHM_RDONLY);
    
    printf("Read: %s\n", shm);
    
    shmdt(shm);
    shmctl(shmid, IPC_RMID, NULL);  // Clean up
    return 0;
}
```

## Message Queues

```c
#include <sys/msg.h>

// Create
int msqid = msgget(key, IPC_CREAT | 0666);

// Send message
struct msgbuf {
    long mtype;
    char mtext[100];
} msg;

msg.mtype = 1;
sprintf(msg.mtext, "Hello");
msgsnd(msqid, &msg, sizeof(msg.mtext), 0);

// Receive
msgrcv(msqid, &msg, sizeof(msg.mtext), 1, 0);
```
